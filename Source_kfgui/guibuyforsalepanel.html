        GUILabel(NewForSalePanel.Controls[0]).ShadowOffsetY = 1.5;

        NewForSalePanel.Controls[2].bVisible = false;
        NewForSalePanel.Controls[2].bAcceptsInput = false;

        ForSaleBodyPanelArray[0] = NewForSalePanel;
